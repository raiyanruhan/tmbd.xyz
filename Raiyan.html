import tkinter as tk
import tkinter as tk

class FunnyCopyCard(tk.Frame):
    def __init__(self, master):
        super().__init__(master, bg="#fff", bd=2, relief=tk.RIDGE)
        self.pack(padx=20, pady=20)
        tk.Label(self, text="💻 Frontend Developer", font=("Arial", 16, "bold"), bg="#fff", fg="#044443").pack(anchor="w")
        tk.Label(self, text="Select any text below and it copies automatically! 😁", font=("Arial", 11), bg="#fff", fg="#228B7D").pack(anchor="w", pady=(5,0))
        self.text = tk.Text(self, height=4, width=40, font=("JetBrains Mono", 11), wrap="word", bg="#f5f7fa")
        self.text.insert("1.0", "Try selecting any part of this text.\nNo Ctrl+C needed!")
        self.text.pack(pady=(10,0))
        self.text.bind("<<Selection>>", self.auto_copy)
        self.copied_label = tk.Label(self, text="", font=("Arial", 10), fg="#66ADA5", bg="#fff")
        self.copied_label.pack()

    def auto_copy(self, event):
        try:
            selected = self.text.get(tk.SEL_FIRST, tk.SEL_LAST)
            self.text.clipboard_clear()
            self.text.clipboard_append(selected)
            self.copied_label.config(text="Copied!")
        except tk.TclError:
            pass

if __name__ == "__main__":
    root = tk.Tk()
    root.title("Funny Auto-Copy Card")
    FunnyCopyCard(root)
    root.mainloop()
